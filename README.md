# Install

requirement:
  python: 2.7 (2.7.6 suggested)

- Install virtualenv

  pip install virtualenv

- Setup env

  virtualenv env

- Activate env

  source env/bin/activate

- Install the requirements

  pip install -r requirements.txt

# Run Django Web App

- Activate env

  source env/bin/activate

- Run server

  cd mysite
  python manage.py runserver

  check http://127.0.0.1:8000/
